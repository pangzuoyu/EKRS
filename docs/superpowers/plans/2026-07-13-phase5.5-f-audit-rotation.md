# Phase 5.5 F: Audit Rotation + /healthz Filter

> **For agentic workers:** REQUIRED SUB-SKILL: superpowers:subagent-driven-development

**Goal**: Bound audit.log size (100MB × 5 gzip) + skip /healthz audit events

**Architecture**: RotatingFileHandler w/ gzip rotator + on_rollover callback → AuditIndex rebuild. ContextVar `_skip_audit` set by middleware for /healthz.

**Tech Stack**: Python stdlib `logging.handlers.RotatingFileHandler` (namer/rotator), `contextvars`, FastAPI middleware

## Global Constraints

- 15 audit event names/schemas UNCHANGED (R-equivalent)
- 7 Iron Rules (R1-R7) UNCHANGED
- Hard delete per D3 — no BC shim
- CLAUDE.md: "permanent, never disabled" → "permanent, size-bounded by rotation (100MB × 5 gzip)"
- Rotation: `RotatingFileHandler(maxBytes=100MB, backupCount=5, namer=gzip_namer, rotator=gzip_rotator)`
- /healthz: ALL audit events skipped via `get_skip_audit() == True`
- AuditIndex rebuild scans ONLY current audit.log (not .gz); uses existing `.build()` method
- on_rollover callback wired in main.py lifespan (AuditWriter created before AuditIndex; callback closure reads `_audit_index` lazily)

```python
# ekrs_rag/observability/audit_handler.py
"""RotatingFileHandler with gzip rotator + on_rollover callback.

Used by AuditWriter to bound audit.log size. The on_rollover hook lets
AuditIndex rebuild its byte-offset index after the file rotates (old
offsets point into the now-renamed audit.log.1.gz and become invalid).
"""
from __future__ import annotations

import gzip
import logging
import os
import shutil
from logging.handlers import RotatingFileHandler


def gzip_namer(name: str) -> str:
    """Append .gz to rotated file names."""
    return name + ".gz"


def gzip_rotator(source: str, dest: str) -> None:
    """Gzip source file into dest during rollover."""
    with open(source, "rb") as f_in, gzip.open(dest, "wb") as f_out:
        shutil.copyfileobj(f_in, f_out)
    os.remove(source)


class RebuildingRotatingFileHandler(RotatingFileHandler):
    """RotatingFileHandler that invokes on_rollover after each rotation."""

    def __init__(
        self, filename, mode="a", maxBytes=0, backupCount=0,
        encoding=None, delay=False, errors=None, on_rollover=None,
    ):
        super().__init__(
            filename, mode, maxBytes, backupCount,
            encoding, delay, errors,
        )
        self._on_rollover = on_rollover

    def doRollover(self):
        super().doRollover()
        if self._on_rollover is not None:
            try:
                self._on_rollover()
            except Exception:
                logging.getLogger("ekrs.audit.rollover").exception(
                    "on_rollover callback failed"
                )
```

import gzip
import os
import tempfile
from pathlib import Path

from ekrs_rag.observability.audit_handler import (
    RebuildingRotatingFileHandler,
    gzip_namer,
    gzip_rotator,

def test_gzip_namer_appends_gz():
    assert gzip_namer("audit.log.1") == "audit.log.1.gz"
    assert gzip_namer("/var/log/x.5") == "/var/log/x.5.gz"


def test_gzip_rotator_compresses_file(tmp_path):
    src = tmp_path / "src.txt"
    src.write_text("hello world\n" * 100)
    dst = tmp_path / "dst.txt.gz"
    gzip_rotator(str(src), str(dst))
    assert not src.exists()
    assert dst.exists()
    with gzip.open(str(dst), "rt") as f:
        assert f.read() == "hello world\n" * 100


def test_rollover_triggers_callback(tmp_path):
    log = tmp_path / "app.log"
    calls = []
    h = RebuildingRotatingFileHandler(
        str(log), maxBytes=10, backupCount=2,
        namer=gzip_namer, rotator=gzip_rotator,
        on_rollover=lambda: calls.append(1),
    )
    # Force rollover past maxBytes
    for _ in range(20):
        h.emit(type("R", (), {"name": "x", "msg": "x"*20, "levelno": 0, "levelname": "I"})())
    h.close()
    assert len(calls) >= 1
